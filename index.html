// PID Core Balancing Logic Snippet
 #include 
#include 

MPU6050 mpu;
int16_t ax, ay, az, gx, gy, gz;

// PID Constants - These MUST be tuned for your specific robot
double Kp = 25, Ki = 120, Kd = 1.5;
double setPoint = 0; // The angle where your robot is perfectly balanced

double input, output, lastInput;
double ITerm, lastTime;
int sampleTime = 10; // 10ms

// Motor Pins (L298N example)
int ENA = 3, IN1 = 4, IN2 = 5;
int ENB = 6, IN3 = 7, IN4 = 8;

void setup() {
  Wire.begin();
  mpu.initialize();
  pinMode(ENA, OUTPUT); pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT); pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
}

void loop() {
  unsigned long now = millis();
  int timeChange = (now - lastTime);

  if(timeChange >= sampleTime) {
    // 1. Get Angle from MPU6050 (Simplified)
    mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
    input = atan2(ax, az) * 180 / PI; // Simple tilt angle calculation

    // 2. PID Calculation
    double error = setPoint - input;
    ITerm += (Ki * error);
    double dInput = (input - lastInput);
    
    output = Kp * error + ITerm - Kd * dInput;

    // 3. Drive Motors
    driveMotors(output);

    // Save variables for next loop
    lastInput = input;
    lastTime = now;
  }
}

void driveMotors(double pidValue) {
  int speed = constrain(abs(pidValue), 0, 255);
  if (pidValue > 0) { // Forward
    digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
    digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
  } else { // Backward
    digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
    digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
  }
  analogWrite(ENA, speed);
  analogWrite(ENB, speed);
}

// --- EEPROM Addresses ---
const int addr_Layer1 = 10; // Address for User PIN
const int addr_Layer2 = 20; // Address for Master/Admin PIN

// --- State Variables ---
char userPIN[7] = "123456"; // Default User PIN
char adminKey[7] = "654321"; // Default Master Key
char inputBuffer[7];
byte inputIndex = 0;
bool isLocked = true;
int failedAttempts = 0;

void setup() {
  lockServo.attach(9);
  pinMode(8, OUTPUT); // Buzzer pin
  
  lcd.init();
  lcd.backlight();
  
  // Initialize or Load PINs
  initializeEEPROM();
  loadPINsFromEEPROM();

  lockSafe();
  displayHome();
}

void loop() {
  char key = keypad.getKey();

  if (key) {
    tone(8, 1000, 50); // Key press feedback
    
    if (key == '#') {
      processInput();
    } 
    else if (key == '*') {
      clearInput();
    } 
    else {
      if (isLocked && inputIndex < 6) {
        inputBuffer[inputIndex] = key;
        lcd.setCursor(inputIndex + 5, 1);
        lcd.print("*");
        inputIndex++;
      }
    }
  }
}

void initializeEEPROM() {
  if (EEPROM.read(addr_Layer1) == 0xFF) { 
    // EEPROM is blank, write default passwords
    saveToEEPROM(addr_Layer1, userPIN);
    saveToEEPROM(addr_Layer2, adminKey);
  }
}

void saveToEEPROM(int address, const char* pinString) {
  for (int i = 0; i < 6; i++) {
    EEPROM.write(address + i, pinString[i]);
  }
}

void loadPINsFromEEPROM() {
  for (int i = 0; i < 6; i++) {
    userPIN[i] = EEPROM.read(addr_Layer1 + i);
    adminKey[i] = EEPROM.read(addr_Layer2 + i);
  }
  userPIN[6] = '\0';
  adminKey[6] = '\0';
}

void processInput() {
  inputBuffer[inputIndex] = '\0'; // Null-terminate the input
  
  if (isLocked) {
    if (strcmp(inputBuffer, userPIN) == 0) {
      grantAccess("Layer 1 Verified", 90); // Unlock servo to 90 degrees
    } 
    else if (strcmp(inputBuffer, adminKey) == 0) {
      grantAccess("Admin Override", 90);
    } 
    else {
      failedAttempts++;
      handleFailedAttempt();
    }
  } else {
    // If unlocked, treat any valid input as a request to lock
    if (strcmp(inputBuffer, userPIN) == 0 || strcmp(inputBuffer, adminKey) == 0) {
      lockSafe();
    } else {
      lcd.clear();
      lcd.print("Invalid Entry");
      delay(1000);
      displayHome();
    }
  }
  clearInput();
}

void grantAccess(const char* message, int angle) {
  lcd.clear();
  lcd.print("Access Granted");
  lcd.setCursor(0, 1);
  lcd.print(message);
  
  lockServo.write(angle); // Rotate servo
  digitalWrite(8, HIGH); // Beep
  delay(500);
  digitalWrite(8, LOW);
  
  isLocked = false;
  failedAttempts = 0;
  delay(1500);
  
  lcd.clear();
  lcd.print("Enter to Lock");
}

void handleFailedAttempt() {
  if (failedAttempts >= 3) {
    lcd.clear();
    lcd.print("SYSTEM LOCKED");
    // Trigger lockdown for 15 seconds
    for (int i = 15; i > 0; i--) {
      lcd.setCursor(14, 1);
      lcd.print(i);
      tone(8, 2000, 500);
      delay(1000);
    }
    failedAttempts = 0;
  } else {
    lcd.clear();
    lcd.print("Access Denied!");
    lcd.setCursor(0, 1);
    lcd.print("Attempts: ");
    lcd.print(failedAttempts);
    digitalWrite(8, HIGH); // Error Buzzer
    delay(1000);
    digitalWrite(8, LOW);
  }
  displayHome();
}

void lockSafe() {
  lcd.clear();
  lcd.print("Safe Locked");
  lockServo.write(0); // Rotate servo back to 0 degrees
  isLocked = true;
  displayHome();
}

void displayHome() {
  lcd.clear();
  lcd.print("Enter PIN:");
  lcd.setCursor(5, 1);
}

void clearInput() {
  memset(inputBuffer, 0, sizeof(inputBuffer));
  inputIndex = 0;
  displayHome();
}

// Basic multiplexing display function
void displayAudio(int level) {
  for (int l = 0; l < 4; l++) {
    // Turn off all layers
    for (int i = 0; i < 4; i++) digitalWrite(layers[i], LOW);

    // If current layer is within audio level, light up columns
    if (l < level) {
      for (int c = 0; c < 16; c++) digitalWrite(columns[c], LOW); // Low = ON for common cathode
      digitalWrite(layers[l], HIGH); // Activate layer
      delay(2); // Short delay for persistence of vision
    }
  }

  digitalWrite(ignitionRelay, HIGH); // Turn OFF Ignition 
  Serial.println("Vehicle Off");

  return finger.fingerID;

int stepSize = 1;               // Degrees to move servo per loop
int maxAngleH = 160;            // Horizontal limits
int minAngleH = 20;
int maxAngleV = 160;            // Vertical limits
int minAngleV = 20;
int sleepCycles = 15;           // 15 x 8s = ~2 mins sleep between logs

  Wire.begin();
  
  // Attach Servos
  servoH.attach(ServoHorizPin);
  servoV.attach(ServoVertPin);
  
  // Set initial positions
  servoH.write(posH);
  servoV.write(posV);
  
  // Initialize RTC
  if (!rtc.begin()) {
    while (1); // Halt if RTC missing
  }
  
  // Initialize SD Card
  if (!SD.begin(ChipSelect)) {
    while (1); // Halt if SD missing
  }

  // Setup Watchdog Timer to wake every 8s
  // (We'll loop this in deep sleep to save power)
  setupWatchdog();

void loop() {
  // 1. Wake Up & Initialize Peripherals
  power_all_enable(); // Turn everything on
  servoH.attach(ServoHorizPin);
  servoV.attach(ServoVertPin);

  // 2. Track Sun
  sunTracker();

  // 3. Log Data to SD Card
  logDataToSD();

  // 4. Power down and park servos
  servoH.detach();
  servoV.detach();

  // 5. Enter Deep Sleep (Saves μA of power)
  for (int i = 0; i < sleepCycles; i++) {
    LowPower.powerDown(SLEEP_8S, ADC_OFF, BOD_OFF);
  }
}

void sunTracker() {
  int top = analogRead(LDR_Top);
  int bottom = analogRead(LDR_Bottom);
  int left = analogRead(LDR_Left);
  int right = analogRead(LDR_Right);

  int avgV = (top + bottom) / 2;
  int avgH = (left + right) / 2;

  // Horizontal Tracking
  if (abs(avgH - bottom) > tolerance) {
    if (left > right) {
      posH += stepSize;
    } else {
      posH -= stepSize;
    }
  }

  // Vertical Tracking
  if (abs(avgV - left) > tolerance) {
    if (top > bottom) {
      posV += stepSize;
    } else {
      posV -= stepSize;
    }
  }

  // Constrain bounds
  posH = constrain(posH, minAngleH, maxAngleH);
  posV = constrain(posV, minAngleV, maxAngleV);

  // Move Servos
  servoH.write(posH);
  servoV.write(posV);
  
  delay(15); // Allow time for servos to reach position
}

void logDataToSD() {
  DateTime now = rtc.now();
  
  // Open file (or create it)
  logFile = SD.open("tracker.csv", FILE_WRITE);
  
  if (logFile) {
    // Timestamp
    logFile.print(now.year(), DEC);
    logFile.print('/');
    logFile.print(now.month(), DEC);
    logFile.print('/');
    logFile.print(now.day(), DEC);
    logFile.print(" ");
    logFile.print(now.hour(), DEC);
    logFile.print(':');
    logFile.print(now.minute(), DEC);
    logFile.print(':');
    logFile.print(now.second(), DEC);
    logFile.print(",");
    
    // Sensor values
    logFile.print(analogRead(LDR_Top));
    logFile.print(",");
    logFile.print(analogRead(LDR_Bottom));
    logFile.print(",");
    logFile.print(analogRead(LDR_Left));
    logFile.print(",");
    logFile.print(analogRead(LDR_Right));
    logFile.print(",");

    logFile.print(posH);
    logFile.print(",");
    logFile.println(posV);
    
    logFile.close();
  }
}

void setupWatchdog() {

  // Set Watchdog Interrupt Mode with an 8-Second interval
  WDTCSR = (1 << WDIE) | (1 << WDP3) | (1 << WDP0);
}

// AES-128 decryption helper
String decryptToken(String encryptedHex) {
  int len = encryptedHex.length();
  if (len % 2 != 0) return ""; 

  unsigned char encryptedBytes[16];
  unsigned char decryptedBytes[16];

  for(int i = 0; i < 16; i++) {
    sscanf(encryptedHex.substring(i*2, i*2+2).c_str(), "%hhx", &encryptedBytes[i]);
  }

  mbedtls_aes_context aes;
  mbedtls_aes_init(&aes);
  mbedtls_aes_setkey_dec(&aes, aes_key, 128);
  mbedtls_aes_crypt_ecb(&aes, MBEDTLS_AES_DECRYPT, encryptedBytes, decryptedBytes);
  mbedtls_aes_free(&aes);

  char output[17];
  memcpy(output, decryptedBytes, 16);
  output[16] = '\0';
  return String(output);
}

  Serial.println("Decrypting and verifying: " + inputString);
  
  String decrypted = decryptToken(inputString);
  
  if (decrypted.startsWith("ACCESS_")) {
    // Extract timestamp or ID for anti-replay (e.g., ACCESS_1243542)
    Serial.println("Access Granted! Decrypted payload: " + decrypted);
    triggerUnlock();
  } else {
    Serial.println("Invalid or Plagiarized Token");
    triggerAlarm();
  }
}

void triggerUnlock() {
  digitalWrite(redLed, LOW);
  digitalWrite(greenLed, HIGH);
  digitalWrite(relayPin, LOW); // Unlock
  tone(buzzerPin, 2000, 200);
  delay(500);
  tone(buzzerPin, 2000, 200);
  
  delay(5000); // Wait 5 seconds
  
  digitalWrite(relayPin, HIGH); // Lock again
  digitalWrite(redLed, HIGH);
  digitalWrite(greenLed, LOW);
}

void triggerAlarm() {
  for(int i=0; i<3; i++) {
    digitalWrite(redLed, HIGH);
    tone(buzzerPin, 800, 300);
    delay(300);
    digitalWrite(redLed, LOW);
    delay(300);
  }
  digitalWrite(redLed, HIGH);
}

// Parse incoming BLE packets (Format expected: "A90 B45 C180 D0")
void parseAndMove(String command) {
  int angles[4];
  int index = 0;
  
  // Simple delimiter parser to extract 4 angles
  char* ptr = strtok((char*)command.c_str(), " ");
  while (ptr != NULL && index < 4) {
    angles[index++] = atoi(ptr + 1); // Skip the letter identifier (e.g., 'A')
    ptr = strtok(NULL, " ");
  }

  // Write the parsed angles to corresponding servos
  for (int i = 0; i < index; i++) {
    // Constrain to valid servo range (0-180 degrees)
    int validAngle = constrain(angles[i], 0, 180);
    servos[i].write(validAngle);
  }
}

  // Create the BLE Server
