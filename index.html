// PID Core Balancing Logic Snippet
 #include 
#include 

MPU6050 mpu;
int16_t ax, ay, az, gx, gy, gz;

// PID Constants - These MUST be tuned for your specific robot
double Kp = 25, Ki = 120, Kd = 1.5;
double setPoint = 0; // The angle where your robot is perfectly balanced

double input, output, lastInput;
double ITerm, lastTime;
int sampleTime = 10; // 10ms

// Motor Pins (L298N example)
int ENA = 3, IN1 = 4, IN2 = 5;
int ENB = 6, IN3 = 7, IN4 = 8;

void setup() {
  Wire.begin();
  mpu.initialize();
  pinMode(ENA, OUTPUT); pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT); pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
}

void loop() {
  unsigned long now = millis();
  int timeChange = (now - lastTime);

  if(timeChange >= sampleTime) {
    // 1. Get Angle from MPU6050 (Simplified)
    mpu.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
    input = atan2(ax, az) * 180 / PI; // Simple tilt angle calculation

    // 2. PID Calculation
    double error = setPoint - input;
    ITerm += (Ki * error);
    double dInput = (input - lastInput);
    
    output = Kp * error + ITerm - Kd * dInput;

    // 3. Drive Motors
    driveMotors(output);

    // Save variables for next loop
    lastInput = input;
    lastTime = now;
  }
}

void driveMotors(double pidValue) {
  int speed = constrain(abs(pidValue), 0, 255);
  if (pidValue > 0) { // Forward
    digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
    digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
  } else { // Backward
    digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
    digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
  }
  analogWrite(ENA, speed);
  analogWrite(ENB, speed);
}

#include

// --- Security Configuration ---
const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {5, 4, 3, 2}; // Connect to the row pinouts of the keypad
byte colPins[COLS] = {A1, A0, 7, 6}; // Connect to the column pinouts of the keypad

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);
LiquidCrystal_I2C lcd(0x27, 16, 2);
Servo lockServo;

// --- EEPROM Addresses ---
const int addr_Layer1 = 10; // Address for User PIN
const int addr_Layer2 = 20; // Address for Master/Admin PIN

// --- State Variables ---
char userPIN[7] = "123456"; // Default User PIN
char adminKey[7] = "654321"; // Default Master Key
char inputBuffer[7];
byte inputIndex = 0;
bool isLocked = true;
int failedAttempts = 0;

void setup() {
  lockServo.attach(9);
  pinMode(8, OUTPUT); // Buzzer pin
  
  lcd.init();
  lcd.backlight();
  
  // Initialize or Load PINs
  initializeEEPROM();
  loadPINsFromEEPROM();

  lockSafe();
  displayHome();
}

void loop() {
  char key = keypad.getKey();

  if (key) {
    tone(8, 1000, 50); // Key press feedback
    
    if (key == '#') {
      processInput();
    } 
    else if (key == '*') {
      clearInput();
    } 
    else {
      if (isLocked && inputIndex < 6) {
        inputBuffer[inputIndex] = key;
        lcd.setCursor(inputIndex + 5, 1);
        lcd.print("*");
        inputIndex++;
      }
    }
  }
}

void initializeEEPROM() {
  if (EEPROM.read(addr_Layer1) == 0xFF) { 
    // EEPROM is blank, write default passwords
    saveToEEPROM(addr_Layer1, userPIN);
    saveToEEPROM(addr_Layer2, adminKey);
  }
}

void saveToEEPROM(int address, const char* pinString) {
  for (int i = 0; i < 6; i++) {
    EEPROM.write(address + i, pinString[i]);
  }
}

void loadPINsFromEEPROM() {
  for (int i = 0; i < 6; i++) {
    userPIN[i] = EEPROM.read(addr_Layer1 + i);
    adminKey[i] = EEPROM.read(addr_Layer2 + i);
  }
  userPIN[6] = '\0';
  adminKey[6] = '\0';
}

void processInput() {
  inputBuffer[inputIndex] = '\0'; // Null-terminate the input
  
  if (isLocked) {
    if (strcmp(inputBuffer, userPIN) == 0) {
      grantAccess("Layer 1 Verified", 90); // Unlock servo to 90 degrees
    } 
    else if (strcmp(inputBuffer, adminKey) == 0) {
      grantAccess("Admin Override", 90);
    } 
    else {
      failedAttempts++;
      handleFailedAttempt();
    }
  } else {
    // If unlocked, treat any valid input as a request to lock
    if (strcmp(inputBuffer, userPIN) == 0 || strcmp(inputBuffer, adminKey) == 0) {
      lockSafe();
    } else {
      lcd.clear();
      lcd.print("Invalid Entry");
      delay(1000);
      displayHome();
    }
  }
  clearInput();
}

void grantAccess(const char* message, int angle) {
  lcd.clear();
  lcd.print("Access Granted");
  lcd.setCursor(0, 1);
  lcd.print(message);
  
  lockServo.write(angle); // Rotate servo
  digitalWrite(8, HIGH); // Beep
  delay(500);
  digitalWrite(8, LOW);
  
  isLocked = false;
  failedAttempts = 0;
  delay(1500);
  
  lcd.clear();
  lcd.print("Enter to Lock");
}

void handleFailedAttempt() {
  if (failedAttempts >= 3) {
    lcd.clear();
    lcd.print("SYSTEM LOCKED");
    // Trigger lockdown for 15 seconds
    for (int i = 15; i > 0; i--) {
      lcd.setCursor(14, 1);
      lcd.print(i);
      tone(8, 2000, 500);
      delay(1000);
    }
    failedAttempts = 0;
  } else {
    lcd.clear();
    lcd.print("Access Denied!");
    lcd.setCursor(0, 1);
    lcd.print("Attempts: ");
    lcd.print(failedAttempts);
    digitalWrite(8, HIGH); // Error Buzzer
    delay(1000);
    digitalWrite(8, LOW);
  }
  displayHome();
}

void lockSafe() {
  lcd.clear();
  lcd.print("Safe Locked");
  lockServo.write(0); // Rotate servo back to 0 degrees
  isLocked = true;
  displayHome();
}

void displayHome() {
  lcd.clear();
  lcd.print("Enter PIN:");
  lcd.setCursor(5, 1);
}

void clearInput() {
  memset(inputBuffer, 0, sizeof(inputBuffer));
  inputIndex = 0;
  displayHome();
}
